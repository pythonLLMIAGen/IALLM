# ====================================
# TEST
# ====================================
def test_model(model_name, model_config, prompt_data):
    start = time.time()
    try:
        response = requests.post(
            model_config["url"],
            headers={
                "Authorization": f"Bearer {model_config['api_key']}",
                "Content-Type": "application/json"
            },
            json={
                "model": model_config["model_id"],
                "messages": [
                    {"role": "system", "content": SYSTEM_PROMPT},
                    {"role": "user", "content": prompt_data["prompt"]}
                ],
                "max_tokens": 500
            },
            proxies=PROXIES,
            timeout=120
        )
        latency = time.time() - start
        data = response.json()

        if response.status_code == 200:
            content = data["choices"][0]["message"]["content"]
            tokens = data["usage"]["total_tokens"]
            return {
                "model": model_name,
                "id": prompt_data["id"],
                "categorie": prompt_data["categorie"],
                "latency": round(latency, 2),
                "tokens": tokens,
                "status": "OK",
                "response": content
            }
        else:
            return {
                "model": model_name,
                "id": prompt_data["id"],
                "categorie": prompt_data["categorie"],
                "latency": round(latency, 2),
                "tokens": 0,
                "status": f"ERREUR {response.status_code}",
                "response": str(data)
            }

    except Exception as e:
        latency = time.time() - start
        return {
            "model": model_name,
            "id": prompt_data["id"],
            "categorie": prompt_data["categorie"],
            "latency": round(latency, 2),
            "tokens": 0,
            "status": "EXCEPTION",
            "response": str(e)
        }

                print(f"  Réponse: {result['response'][:300]}...")

        # Résumé par modèle
        ok = [r for r in model_results if r["status"] == "OK"]
        if ok:
            avg_latency = sum(r["latency"] for r in ok) / len(ok)

            print(f"\nRésumé {model_name}:")
            print(f"  Succès      : {len(ok)}/{len(PROMPTS)}")

    # ====================================

    latences = {}
    for model_name in MODELS.keys():
        ok = [r for r in all_results if r["model"] == model_name and r["status"] == "OK"]
        if ok:
            avg_lat = sum(r["latency"] for r in ok) / len(ok)
            avg_tok = sum(r["tokens"] for r in ok) / len(ok)
            latences[model_name] = avg_lat
            print(f"{model_name:10} | Latence moy: {avg_lat:.2f}s | Tokens moy: {avg_tok:.0f} | Succès: {len(ok)}/{len(PROMPTS)}")

    if latences:
        vainqueur = min(latences, key=latences.get)
        print(f"\nVainqueur latence : {vainqueur} 🏆")

    # ====================================
    # EXPORT CSV
    # ====================================

        writer = csv.DictWriter(f, fieldnames=["model", "id", "categorie", "latency", "tokens", "status", "response"])
        writer.writeheader()
        writer.writerows(all_results)

    print(f"\nRésultats exportés : {filename}")
