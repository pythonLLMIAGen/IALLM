data "aws_iam_policy_document" "sam_appsync_logs_assume" {
  statement {
    effect  = "Allow"
    actions = ["sts:AssumeRole"]
    principals {
      type        = "Service"
      identifiers = ["appsync.amazonaws.com"]
    }
  }
}

resource "aws_iam_role" "sam_appsync_logs_role" {
  name               = "${local.sam_open_deep_research_appsync_name}-logs-role"
  assume_role_policy = data.aws_iam_policy_document.sam_appsync_logs_assume.json
  tags               = merge(local.xxxx_base_tags, { Name = "${local.sam_open_deep_research_appsync_name}-logs-role" })
}

data "aws_iam_policy_document" "sam_appsync_logs_policy" {
  statement {
    effect = "Allow"
    actions = [
      "logs:CreateLogGroup",
      "logs:CreateLogStream",
      "logs:PutLogEvents"
    ]
    resources = [
      "${aws_cloudwatch_log_group.sam_open_deep_research_appsync_logs.arn}:*"
    ]
  }
}

resource "aws_iam_policy" "sam_appsync_logs_policy" {
  name   = "${local.sam_open_deep_research_appsync_name}-logs-policy"
  policy = data.aws_iam_policy_document.sam_appsync_logs_policy.json
  tags   = merge(local.xxxx_base_tags, { Name = "${local.sam_open_deep_research_appsync_name}-logs-policy" })
}

resource "aws_iam_role_policy_attachment" "sam_appsync_logs_attach" {
  role       = aws_iam_role.sam_appsync_logs_role.name
  policy_arn = aws_iam_policy.sam_appsync_logs_policy.arn
}
