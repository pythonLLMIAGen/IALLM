# ====================================
# APPEL LLM
# ====================================
def call_llm(messages, max_tokens=500):
    response = requests.post(
        LITELLM_URL,
        headers={
            "Authorization": f"Bearer {API_KEY}",
            "Content-Type": "application/json"
        },
        json={
            "model": "eu.anthropic.claude-sonnet-4-5-20250929-v1:0",
            "messages": messages,
            "max_tokens": max_tokens
        },
        timeout=120
    )
    if response.status_code == 200:
        return response.json()["choices"][0]["message"]["content"]
    return None

# ====================================
# EVALUATION AUTOMATIQUE
# ====================================
def evaluer_reponse(question, reponse):
    import json
    prompt = EVALUATION_PROMPT.format(
        question=question,
        reponse=reponse
    )
    try:
        result = call_llm([{"role": "user", "content": prompt}], max_tokens=200)
        if result:
            # Nettoie le JSON
            result = result.strip()
            if "```" in result:
                result = result.split("```")[1].replace("json", "").strip()
            data = json.loads(result)
            return {
                "score_empathie": data.get("empathie", 0),
                "score_pertinence": data.get("pertinence", 0),
                "score_clarte": data.get("clarte", 0),
                "score_global": data.get("score_global", 0),
                "commentaire": data.get("commentaire", "")
            }
    except Exception as e:
        print(f"  Erreur évaluation: {e}")
    return {
        "score_empathie": 0,
        "score_pertinence": 0,
        "score_clarte": 0,
        "score_global": 0,
        "commentaire": "Erreur évaluation"
    }

# ====================================
# TEST MODELE
# ====================================
def test_model(model_name, model_id, prompt_data):
    start = time.time()
    try:
        response = requests.post(
            LITELLM_URL,
            headers={
                "Authorization": f"Bearer {API_KEY}",
                "Content-Type": "application/json"
            },
            json={
                "model": model_id,
                "messages": [
                    {"role": "system", "content": SYSTEM_PROMPT},
                    {"role": "user", "content": prompt_data["prompt"]}
                ],
                "max_tokens": 500
            },
            timeout=120
        )
        latency = time.time() - start
        data = response.json()

        if response.status_code == 200:
            content = data["choices"][0]["message"]["content"]
            tokens = data["usage"]["total_tokens"]

            # Evaluation automatique
            print(f"  Evaluation en cours...", end=" ", flush=True)
            scores = evaluer_reponse(prompt_data["prompt"], content)

            return {
                "model": model_name,
                "id": prompt_data["id"],
                "categorie": prompt_data["categorie"],
                "latency": round(latency, 2),
                "tokens": tokens,
                "status": "OK",
                "score_empathie": scores["score_empathie"],
                "score_pertinence": scores["score_pertinence"],
                "score_clarte": scores["score_clarte"],
                "score_global": scores["score_global"],
                "commentaire": scores["commentaire"],
                "response": content
            }
        else:
            return {
                "model": model_name,
                "id": prompt_data["id"],
                "categorie": prompt_data["categorie"],
                "latency": round(latency, 2),
                "tokens": 0,
                "status": f"ERREUR {response.status_code}",
                "score_empathie": 0,
                "score_pertinence": 0,
                "score_clarte": 0,
                "score_global": 0,
                "commentaire": "",
                "response": str(data)
            }

    except Exception as e:
        latency = time.time() - start
        return {
            "model": model_name,
            "id": prompt_data["id"],
            "categorie": prompt_data["categorie"],
            "latency": round(latency, 2),
            "tokens": 0,
            "status": "EXCEPTION",
            "score_empathie": 0,
            "score_pertinence": 0,
            "score_clarte": 0,
            "score_global": 0,
            "commentaire": "",
            "response": str(e)
        }
